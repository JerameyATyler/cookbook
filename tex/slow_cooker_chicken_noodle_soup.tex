\begin{recipe}
[% 
    preparationtime = {\unit[20]{m}},
    bakingtime={\unit[8]{h}},
    portion = {\portion{5-6}},
    calory={465},
    source = {Food.com I think}
]
{Slow Cooker Chicken Noodle Soup}
    
    \graph
    {% pictures
        small=pic/glass,     % small picture
        big=pic/ingredients  % big picture
    }
    
    \introduction{%
        Chicken noodle soup in the slow cooker. Just like grandma used to make. Throw all 				the junk in the slow cooker and come home to delicousness.
    }
    
    \ingredients{%
        6 cups & low sodium chicken broth\\
        3 & carrots sliced\\
		3 Stalks & celery sliced\\
        2 medium & Onions, chopped\\
        1 clove & garlic\\
        2 & bay leaves\\
        1/2 tsp & dried thyme\\
        4 tsp & salt\\
        1/2 tsp & fresh ground black pepper\\
        1 (3 1/2 lb) & roasting chicken\\
        3 cups & wide egg noodles, uncooked
    }
    
    \preparation{%
        \step In 4-1/2 to 6-quart slow c00ker combine water carrots, celery, onions, garlic, 				bay leaves, thyme, measured salt, and measured pepper.
        \step Place chicken on top of vegetables.
        \step Cover slow cooker with lid and cook on low for 8-10 hours or on high for 4-5 						hours.
        \step Transfer chicken to cutting board. Discard bay leaves. Add noodles to slow 						cooker and cooked covered for an additional 20 minutes.
        \step While noodles cook remove and discard the skin, fat, and bones from the 							chicken and shred the meat.
        \step Skim fat from soup and discard. Return chicken to soup to serve.
    }
    
    \suggestion[Notes]
    {%
        \begin{itemize}
        	\item Chicken broth can be substituted with water
        	\item Omit the measured salt if you use regular sodium chicken broth
        	\item 2-3 chicken breasts can be used instead of a roasting chicken. Easier to 							.shred
        	\item If you cook them too long the noodles will disolve and make your soup 							thick.
        \end{itemize}
    }
    
\end{recipe}
